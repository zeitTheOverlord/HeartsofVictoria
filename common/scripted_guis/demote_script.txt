se_demote = {
    scope = character
    effect = {
		add_commander_rank = -1
		interest_group = { 
			add_modifier = {
				name = Demotion
				years = 5
			}
		}
    }
}
se_total_demote = {
    scope = character
    effect = {
		set_variable = {
			name = demotion_ranks
			value = {
				add = {
					value = commander_rank
					subtract = 1 
				}
			}
		}
		while = {
			count = var:demotion_ranks
			interest_group = { 
				add_modifier = {
					name = Demotion
					years = 5
				}
			}
		}
		set_commander_rank = 1
    }
}
