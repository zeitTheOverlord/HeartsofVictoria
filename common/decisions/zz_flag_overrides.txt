spa_burgundian_cross_enabled = {
	is_shown = {
		is_player = yes
		c:SPA = ROOT
        NOT = {
		    has_variable = spa_burgundian_cross_war_flag_enabled
        }
	}
	
	when_taken = {
		set_variable = spa_burgundian_cross_war_flag_enabled
	}
	
	ai_chance = {
		base = 0
	}
}

}
