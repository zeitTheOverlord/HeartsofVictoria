dp_reshape_order_mng = {
	war_goal = war_reparations
	#allow_negotiated_peace = no
	mirror_war_goal = yes
	requires_interest_marker = no
	is_epic = yes
	texture = "gfx/interface/icons/war_goals/liberate_subject.dds"
	
	#selectable_in_lens = {
	#	always = no
	#}
	
	possible = {	
		global_var:world_tension >= 80		
		has_technology_researched = multilateral_alliances
		NOT = { 
			has_diplomatic_pact = {
				who = scope:target_country
				type = war_reparations
				is_initiator = yes
			}
		}
		scope:target_country = {
			NOT = { 
				has_diplomatic_pact = {
					who = root
					type = war_reparations
					is_initiator = yes
				}
			}
		}
		custom_tooltip = {
			text = qwf_order_challenger
			has_modifier = qwf_order_challenger_modifier
		}
		scope:target_country = {
			custom_tooltip = {
				text = qwf_order_keeper
				has_modifier = qwf_order_keeper_modifier
			}
		}
		custom_tooltip = {
			text = je_great_war_on_going
			NOR = { 
				has_global_variable = world_war_diplomatic_play 
				has_global_variable = minor_world_war_diplomatic_play
			}
		}
		custom_tooltip = {
			text = je_great_war_event_on_going
			NOT = { has_global_variable = world_war_diplomatic_play_event_going }
		}
	}
	
	additional_involvement_trigger = {
		OR = {
			country_rank >= rank_value:great_power
			AND = {
				country_rank = rank_value:major_power
				has_diplomatic_pact = {
					who = scope:initiator
					type = qwf_interest_alliance
				}
			}
			AND = {
				country_rank = rank_value:major_power
				has_diplomatic_pact = {
					who = scope:target_country
					type = qwf_interest_alliance
				}
			}
		}
	}
	
	on_weekly_pulse = {
		if = {
			limit = {
				escalation >= 7
				escalation < 21
			}
			add_escalation = -6
		}
	}
	
	on_war_begins = {}

	on_war_end = {}
}
